/*
1. Pregunta al usuario qué día de la semana es. Si la respuesta es "Sábado" o "Domingo", 
muestra "¡Buen fin de semana!". De lo contrario, muestra "¡Buena semana!".
*/

let respuestaUsuario = prompt("¿Escribe que dia de la semana es hoy?");
console.log(respuestaUsuario);

let sabado = "Sabado";
let domingo = "Domingo"

// opcion 1: Usando condicional ternario con toLowerCase para no distinguir entre mayusculas o minusculas

let mensaje = respuestaUsuario.toLowerCase() === sabado.toLowerCase() || respuestaUsuario.toLowerCase() === domingo.toLowerCase() ? "¡Buen fin de semana!" : "¡Buena semana!";
alert(mensaje);

// opcion 2: Usando if normal con toLowerCase para no distinguir entre mayusculas o minusculas

if(respuestaUsuario.toLowerCase() === sabado.toLowerCase() || respuestaUsuario.toLowerCase() === domingo.toLowerCase()){
    alert("¡Buen fin de semana!");
}else{
    alert("¡Buena semana!")
}
